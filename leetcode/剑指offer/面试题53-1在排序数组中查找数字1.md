**代码 && 解题思路：**

```js
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number}
 */
var search = function(nums, target) {
    let cnt = 0;
    for(let num of nums){
        if(num == target){
            cnt += 1;
        }else if(num > target){         // 后面的都比 target 大，已经没有比较的必要了。
            break;
        }   
    }
    return cnt;
};
```
